class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        //vector<int> value;
        map<int,int> a;
        vector<int> b(2,-1);
        for(int i = 0; i < nums.size() ; i++){
            if(a.count(target - nums[i])){
                b[0] = a[target - nums[i]];
                b[1] = i;
            }
            a[nums[i]] = i;
        }
        return b;
    }
};
